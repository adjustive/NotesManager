87 static QList < QStandardItem * > toStandardItems( RichTextNote * note )
88 {
89  QStandardItem * title = toStandardItem( note );
90  title->setText( note->title() );
91  title->setToolTip( title->text() );
92 
93  QStandardItem * visibility = new QStandardItem();
94  visibility->setText( note->isVisible() ? QTranslator::tr( "yes" ) : QTranslator::tr( "no" ) );
95  visibility->setTextAlignment( Qt::AlignCenter );
96 
97  QStandardItem * created = new QStandardItem();
98  created->setText( note->created().toString( Qt::SystemLocaleLongDate ) );
99 
100  QStandardItem * modified = new QStandardItem();
101  modified->setText( note->modified().toString( Qt::SystemLocaleLongDate ) );
102 
103  QStandardItem * top = new QStandardItem();
104  top->setText( note->isTop() ? QTranslator::tr( "yes" ) : QTranslator::tr( "no" ) );
105  top->setTextAlignment( Qt::AlignCenter );
106 
107  QStandardItem * attachments = new QStandardItem();
108  attachments->setText( QString::number( note->numberOfAttachments() ) );
109  attachments->setTextAlignment( Qt::AlignCenter );
110 
111  return QList < QStandardItem * > () << title << visibility << created << modified << top << attachments;
112 }
114 static QList < QStandardItem * > toStandardItems( const QString & path )
115 {
116  QSettings ini( path + "/" + "settings.ini", QSettings::IniFormat );
117  ini.setIniCodec( "utf8" );
118 
119  QStandardItem * title = new QStandardItem();
120  title->setText( ini.value( "Title" ).toString() );
121  title->setToolTip( title->text() );
122 
123  QStandardItem * visibility = new QStandardItem();
124  visibility->setText( ini.value( "Visible" ).toBool() ? QTranslator::tr( "yes" ) : QTranslator::tr( "no" ) );
125  visibility->setTextAlignment( Qt::AlignCenter );
126 
127  QStandardItem * created = new QStandardItem();
128  created->setText( ini.value( "Created" ).toDateTime().toString( Qt::SystemLocaleLongDate ) );
129 
130  QStandardItem * modified = new QStandardItem();
131  modified->setText( ini.value( "Modified" ).toDateTime().toString( Qt::SystemLocaleLongDate ) );
132 
133  QStandardItem * top = new QStandardItem();
134  top->setText( ini.value( "Top" ).toBool() ? QTranslator::tr( "yes" ) : QTranslator::tr( "no" ) );
135  top->setTextAlignment( Qt::AlignCenter );
136 
137  QStandardItem * attachments = new QStandardItem();
138  attachments->setText( QString::number( QDir( path + "/" + "attach" ).entryList( QDir::Files ).size() ) );
139  attachments->setTextAlignment( Qt::AlignCenter );
140 
141  return QList < QStandardItem * > () << title << visibility << created << modified << top << attachments;
142 }

145 static QStandardItem * findItem( RichTextNote * note, QStandardItemModel * model, int column = 0 )
146 {
147  for ( int row = 0; row < model->rowCount(); row++ )
148  {
149  QStandardItem * item = model->item( row, column );
150  if ( note == toNote( item ) )
151  return item;
152  }
153 
154  return 0;
155 }
